1.Variable Declaration
 	
Declare    
    X Number(3):='10';
    Y Number(3):='20';

Begin
   Dbms_Output.Put_Line(X+Y);
   Dbms_Output.Put_Line(X-Y);
   Dbms_Output.Put_Line(X*Y);
   Dbms_Output.Put_Line(X/Y);
End;
 
Output
  
  30
 -10
  200
  5

Declare    
    X Number(3);
    Y Number(3);

Begin
   Dbms_Output.Put_Line(:X+:Y);
End;
